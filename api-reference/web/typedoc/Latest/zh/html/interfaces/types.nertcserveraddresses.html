Interface NeRtcServerAddresses

私有化服务器相关配置参数。

// 加入房间

  channelName: 'channel163',
  uid: 123,
  token: '<您的token>', // 如关闭了安全模式，则不需要该参数。
  neRtcServerAddresses: {
    channelServer: '',
    statisticsServer: '',
    statisticsWebSocketServer: ',
    roomServer: '',
  },
});
